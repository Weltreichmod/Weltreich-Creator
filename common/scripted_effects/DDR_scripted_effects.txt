### +2.5% Revolution Progress
DDR_inc_revolution_2_5 = {
    custom_effect_tooltip = DDR_inc_revolution_2_5_tt
    if = {
        limit = { check_variable = { DDR_german_revolutionization_var > 97.4 } }
        set_variable = { var = DDR_german_revolutionization_var value = 100 }
    }
    else = {
        add_to_variable = { var = DDR_german_revolutionization_var value = 2.5 }
    }
}
