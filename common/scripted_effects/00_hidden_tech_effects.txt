	custom_override_tooltip = {
		tooltip = {
			localization_key = tooltip_enable_hidden_tech
			TECH = country_culture|tech_precision_strike
			EFFECT = tech_effect|tech_precision_strike
		}
		set_technology = { tech_precision_strike = 1 popup = no }
	}
}
