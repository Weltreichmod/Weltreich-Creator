start_blalance_bf_effect = {
	if = {
		limit = {
			FRA = {
				has_government = ROOT
			}
			NOT = {
				has_country_leader = {
					character = USA_franklin_delano_roosevelt
					ruling_only = yes 
				}
			}
		}
		add_to_variable = { var = option_on_france_var value = 10 }
		else = {
			add_to_variable = { var = option_on_france_var value = -5 }
		}
	}
	if = {
		limit = {
			FRA = {
				has_government = ROOT
			}
			has_country_leader = {
				character = USA_franklin_delano_roosevelt
				ruling_only = yes 
			}
		}
		add_to_variable = { var = option_on_france_var value = 10 }
		else_if = {
			limit = {
				FRA = {
					has_government = fascism
				}
			}
			add_to_variable = { var = option_on_france_var value = -15 }
		}
		else = {
			add_to_variable = { var = option_on_france_var value = -5 }
		}
	}
	if = {
		limit = {
			has_opinion = {
				target = FRA
				value > 70
			}
		}
		add_to_variable = { var = option_on_france_var value = 20 }
	}
	else_if = {
		limit = {
			OR = {
				has_opinion = {
					target = FRA
					value = 70
				}
				has_opinion = {
					target = FRA
					value < 70
				}
			}
			has_opinion = {
				target = FRA
				value > 30
			}
		}
		add_to_variable = { var = option_on_france_var value = 10 }
	}
	else_if = {
		limit = {
			OR = {
				has_opinion = {
					target = FRA
					value = 30
				}
				has_opinion = {
					target = FRA
					value < 30
				}
			}
			has_opinion = {
				target = FRA
				value > 10
			}
		}
		add_to_variable = { var = option_on_france_var value = 5 }
	}
	else_if = {
		limit = {
			OR = {
				has_opinion = {
					target = FRA
					value = 10
				}
				has_opinion = {
					target = FRA
					value < 10
				}
			}
			OR = {
				has_opinion = {
					target = FRA
					value = -10
				}
				has_opinion = {
					target = FRA
					value > -10
				}
			}
		}
		add_to_variable = { var = option_on_france_var value = 0 }
	}
	else_if = {
		limit = {
			OR = {
				has_opinion = {
					target = FRA
					value = -30
				}
				has_opinion = {
					target = FRA
					value > -30
				}
			}
			has_opinion = {
				target = FRA
				value < -10
			}
		}
		add_to_variable = { var = option_on_france_var value = -5 }
	}
	else_if = {
		limit = {
			OR = {
				has_opinion = {
					target = FRA
					value = -60
				}
				has_opinion = {
					target = FRA
					value > -60
				}
			}
			has_opinion = {
				target = FRA
				value < -30
			}
		}
		add_to_variable = { var = option_on_france_var value = -10 }
	}
	else_if = {
		limit = {
			has_opinion = {
				target = FRA
				value < -60
			}
		}
		add_to_variable = { var = option_on_france_var value = -20 }
	}
	if = {
		limit = {
			FRA = { has_offensive_war = yes }
		}
		add_to_variable = { var = option_on_france_var value = -10 }
	}
}
