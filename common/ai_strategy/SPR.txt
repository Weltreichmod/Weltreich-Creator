# Make nobody want to ally Spain during the civil war.
spanish_civil_war_ally_nobody = {
	enable = {
		SPA = { has_war_with = SPD }
		NOT = {
			original_tag = SPR
		}
	}
	abort = {
		NOT = { SPA = { has_war_with = SPD } }
	}

	ai_strategy = {
		type = alliance
		id = SPD
		value = -2000
	}
	ai_strategy = {
		type = alliance
		id = SPA
		value = -2000
	}
}
