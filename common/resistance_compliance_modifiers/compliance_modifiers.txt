compliance_80 = {
	type = core_compliance_modifier
	icon = "GFX_occupation_compliance_modifier_strip:5"
	small_icon = "GFX_occupation_compliance_modifier_small_strip:5"
	threshold = 80
	margin = 2
	
	visible = {
		has_dlc = "La Resistance"
		OR = {
			is_ai = no
			AND = {
				is_ai = yes
				OR = {
					has_government = communism
					is_major = no
					date > 1948.1.1
				}
			}
		}
		NOT = { FROM = { original_tag = event_target:system } }
	}
	enabled = {
		is_available_to_collaboration_government = yes
		has_rule = can_create_collaboration_government
	}
	state_modifier = {
	}
	
	on_enable = {
		country_event = { id = occupied_countries.1 days = 2 }
	}
}

compliance_integration_60 = {
	type = core_compliance_modifier
	icon = "GFX_occupation_compliance_modifier_strip:5"
	small_icon = "GFX_occupation_compliance_modifier_small_strip:5"
	threshold = 60
	margin = 2

	visible = {
		FROM = { original_tag = event_target:system }
	}

	on_enable = {
		custom_effect_tooltip = integration_only_on_owned_states
		effect_tooltip = {
			every_owned_state = {
				limit = {
					check_variable = { can_integrate_@ROOT = 60 }
					is_controlled_by = ROOT
					NOT = { is_core_of = ROOT }
				}
				add_core_of = ROOT
			}
		}
	}
}
