			limit = {
				has_country_flag = second_great_war_loser
			}

			if = {
				limit = {

				limit = {
					has_country_flag = second_great_war_loser
					is_subject = no
				}
				add_timed_idea = { idea = war_reparations days = 1000 }
				add_ideas = peace_treaty

				limit = {
					FRA = { has_country_flag = second_great_war_loser }

					retire_country_leader = yes

					}
